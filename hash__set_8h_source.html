73static inline unsigned int rsv_hash_set_hash(const void* data,
74 unsigned int element_size) {
75 const unsigned char* data_key = (const unsigned char*)data;
76 unsigned int hash = 0;
77 unsigned int i;
78
79 for (i = 0; i < element_size; ++i) {
80 hash = (hash * 31) + data_key[i];
81 }
82
83 return hash;

222static inline void rsv_hash_set_push(rsv_hash_set_t* hash_set,
223 const void* data) {
224 unsigned int index;
225 rsv_hash_set_entry_t* entry;
226 rsv_hash_set_entry_t* new_entry;
227
228 if ((float)hash_set->amount / hash_set->capacity > RSV_HASH_SET_LOAD_FACTOR) {
229 rsv_hash_set_resize(hash_set, hash_set->capacity * 2);
230 }

232 index = hash_set->custom_hash_func(data, hash_set->element_size) %
233 hash_set->capacity;
234 entry = hash_set->data[index];
235
236 while (entry) {
237 if (hash_set->custom_compare_func(entry->data, data,
238 hash_set->element_size)) {
239 return;
240 }
241 entry = entry->next;
242 }
243
244 new_entry = (rsv_hash_set_entry_t*)malloc(sizeof(rsv_hash_set_entry_t));
245 new_entry->data = malloc(hash_set->element_size);
246 memcpy(new_entry->data, data, hash_set->element_size);
247 new_entry->next = hash_set->data[index];
248 hash_set->data[index] = new_entry;
249 hash_set->amount++;

258static inline void rsv_hash_set_pop(rsv_hash_set_t* hash_set,

262 hash_set->capacity;
263 rsv_hash_set_entry_t* entry = hash_set->data[index];
264 rsv_hash_set_entry_t* prev = NULL;
265
266 while (entry) {
267 if (hash_set->custom_compare_func(entry->data, data,
268 hash_set->element_size)) {
269 if (prev) {
270 prev->next = entry->next;
271 } else {
272 hash_set->data[index] = entry->next;
273 }
274
275 free(entry->data);
276 free(entry);
277 hash_set->amount--;
278
279 return;
280 }

283 entry = entry->next;
