46static const char* rsv_file_load_text_file(const char* file_path) {
47 FILE* file = fopen(file_path, "r");
48 long length = 0;
49 char* content = "";
50
51 if (!file) {
52 fprintf(stderr, "Failed to open file\n");
53
54 return NULL;
55 }
56
57 fseek(file, 0, SEEK_END);
58 length = ftell(file);
59 fseek(file, 0, SEEK_SET);
60
61 content = (char*)malloc((length + 1) * sizeof(char));
62 fread(content, 1, length, file);
63 content[length] = '\0';
64
65 fclose(file);
66
67 return content;
68}
