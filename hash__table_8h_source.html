80static inline unsigned int rsv_hash_table_hash(const void* data,
81 unsigned int element_size) {
82 const unsigned char* data_key = (const unsigned char*)data;
83 unsigned int hash = 0;
84 unsigned int i;
85
86 for (i = 0; i < element_size; ++i) {
87 hash = (hash * 31) + data_key[i];
88 }

178 unsigned int new_capacity) {
179 unsigned int i;
180 rsv_hash_table_entry_t** new_data = (rsv_hash_table_entry_t**)calloc(

184 rsv_hash_table_entry_t* entry = hash_table->data[i];

186 rsv_hash_table_entry_t* next = entry->next;
187 unsigned int new_index =
188 hash_table->custom_hash_func(entry->key, hash_table->key_size) %

190 entry->next = new_data[new_index];

197 hash_table->data = new_data;
198 hash_table->capacity = new_capacity;

233static inline void rsv_hash_table_push(rsv_hash_table_t* hash_table,
234 const void* key, const void* value) {
235 unsigned int index;
236 rsv_hash_table_entry_t* entry;
237 rsv_hash_table_entry_t* new_entry;
238
239 if ((float)hash_table->amount / hash_table->capacity >
240 RSV_HASH_TABLE_LOAD_FACTOR) {
241 rsv_hash_table_resize(hash_table, hash_table->capacity * 2);
242 }
243
244 index = hash_table->custom_hash_func(key, hash_table->key_size) %
245 hash_table->capacity;
246 entry = hash_table->data[index];
247
248 while (entry) {
249 if (hash_table->custom_compare_func(entry->key, key,
250 hash_table->key_size)) {
251 memcpy(entry->value, value, hash_table->value_size);
252 return;
253 }
254 entry = entry->next;
255 }
256
257 new_entry = (rsv_hash_table_entry_t*)malloc(sizeof(rsv_hash_table_entry_t));
258 new_entry->key = malloc(hash_table->key_size);
259 new_entry->value = malloc(hash_table->value_size);
260 memcpy(new_entry->key, key, hash_table->key_size);
261 memcpy(new_entry->value, value, hash_table->value_size);
262 new_entry->next = hash_table->data[index];
263 hash_table->data[index] = new_entry;
264 hash_table->amount++;
265}

273static inline void rsv_hash_table_pop(rsv_hash_table_t* hash_table,
274 const void* key) {
275 unsigned int index = hash_table->custom_hash_func(key, hash_table->key_size) %
276 hash_table->capacity;
277 rsv_hash_table_entry_t* entry = hash_table->data[index];
278 rsv_hash_table_entry_t* prev = NULL;
279
280 while (entry) {
281 if (hash_table->custom_compare_func(entry->key, key,
282 hash_table->key_size)) {
283 if (prev) {
284 prev->next = entry->next;
285 } else {
286 hash_table->data[index] = entry->next;
287 }
288
289 free(entry->key);
290 free(entry->value);
291 free(entry);
292 hash_table->amount--;
293
294 return;
295 }
296
297 prev = entry;
298 entry = entry->next;
299 }
