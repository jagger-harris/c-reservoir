393static void* rsv_dynamic_array_get(rsv_dynamic_array_t* array,
394 unsigned int index) {
395 if (index >= array->amount) {
396 return NULL;
397 }
398
399 return (void*)((unsigned char*)array->data + index * array->element_size);
400}

402static void rsv_dynamic_array_push(rsv_dynamic_array_t* array,
403 const void* element) {
404 unsigned char* destination;
405
406 if (array->amount >= array->capacity) {
407 array->capacity =
408 (size_t)ceil(array->capacity * RSV_DYNAMIC_ARRAY_GROWTH_AMOUNT);
409 array->data = realloc(array->data, array->capacity * array->element_size);
410 }
411
412 destination =
413 (unsigned char*)array->data + array->amount * array->element_size;
414 memcpy((void*)destination, element, array->element_size);
415 array->amount++;
416}
417
418static void rsv_dynamic_array_pop(rsv_dynamic_array_t* array) {
419 if (array->amount == 0) {
420 return;
421 }
422
423 array->amount--;
424
425 /* Shrink array if number of elements is less than a quarter */
426 if (array->amount < array->capacity / 4) {
427 array->capacity /= 2;
428 array->data = realloc(array->data, array->capacity * array->element_size);
429 }
430}

434static unsigned int rsv_hash_table_hash(const void* data,
435 unsigned int element_size) {
436 const unsigned char* data_key = (const unsigned char*)data;
437 unsigned int hash = 0;
438 unsigned int i;
439
440 for (i = 0; i < element_size; ++i) {
441 hash = (hash * 31) + data_key[i];
442 }
443
444 return hash;
445}

448 unsigned int element_size) {
449 return memcmp(data1, data2, element_size) == 0;

539static void rsv_hash_table_push(rsv_hash_table_t* hash_table, const void* key,
540 const void* value) {
541 unsigned int index;
542 rsv_hash_table_entry_t* entry;
543 rsv_hash_table_entry_t* new_entry;
544
545 if ((float)hash_table->amount / hash_table->capacity >
546 RSV_HASH_TABLE_LOAD_FACTOR) {
547 rsv_hash_table_resize(hash_table, hash_table->capacity * 2);
548 }
549
550 index = hash_table->custom_hash_func(key, hash_table->key_size) %
551 hash_table->capacity;
552 entry = hash_table->data[index];
553
554 while (entry) {
555 if (hash_table->custom_compare_func(entry->key, key,
556 hash_table->key_size)) {
557 memcpy(entry->value, value, hash_table->value_size);
558 return;
559 }
560 entry = entry->next;
561 }
562
563 new_entry = (rsv_hash_table_entry_t*)malloc(sizeof(rsv_hash_table_entry_t));
564 new_entry->key = malloc(hash_table->key_size);
565 new_entry->value = malloc(hash_table->value_size);
566 memcpy(new_entry->key, key, hash_table->key_size);
567 memcpy(new_entry->value, value, hash_table->value_size);
568 new_entry->next = hash_table->data[index];
569 hash_table->data[index] = new_entry;
570 hash_table->amount++;
571}

573static void rsv_hash_table_pop(rsv_hash_table_t* hash_table, const void* key) {
574 unsigned int index = hash_table->custom_hash_func(key, hash_table->key_size) %
575 hash_table->capacity;
576 rsv_hash_table_entry_t* entry = hash_table->data[index];
577 rsv_hash_table_entry_t* prev = NULL;
578
579 while (entry) {
580 if (hash_table->custom_compare_func(entry->key, key,
581 hash_table->key_size)) {
582 if (prev) {
583 prev->next = entry->next;
584 } else {
585 hash_table->data[index] = entry->next;
586 }
587
588 free(entry->key);
589 free(entry->value);
590 free(entry);
591 hash_table->amount--;
592
593 return;
594 }
595
596 prev = entry;
597 entry = entry->next;
598 }
