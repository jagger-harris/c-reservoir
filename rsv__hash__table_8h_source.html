83static inline unsigned int rsv_hash_table_hash(const void* data,
84 unsigned int element_size) {
85 const unsigned char* data_key = (const unsigned char*)data;
86 unsigned int hash = 0;
87 unsigned int i;
88
89 for (i = 0; i < element_size; ++i) {
90 hash = (hash * 31) + data_key[i];
91 }
92
93 return hash;
94}

180static inline void rsv_hash_table_resize(rsv_hash_table_t* hash_table,
181 unsigned int new_capacity) {
182 unsigned int i;
183 rsv_hash_table_entry_t** new_data = (rsv_hash_table_entry_t**)calloc(
184 new_capacity, sizeof(rsv_hash_table_entry_t*));
185
186 for (i = 0; i < hash_table->capacity; ++i) {
187 rsv_hash_table_entry_t* entry = hash_table->data[i];
188 while (entry) {
189 rsv_hash_table_entry_t* next = entry->next;
190 unsigned int new_index =
191 hash_table->custom_hash_func(entry->key, hash_table->key_size) %
192 new_capacity;
193 entry->next = new_data[new_index];
194 new_data[new_index] = entry;
195 entry = next;
196 }
197 }
198
199 free(hash_table->data);
200 hash_table->data = new_data;
201 hash_table->capacity = new_capacity;
202}

236static inline void rsv_hash_table_push(rsv_hash_table_t* hash_table,
237 const void* key, const void* value) {
238 unsigned int index;
239 rsv_hash_table_entry_t* entry;
240 rsv_hash_table_entry_t* new_entry;
241
242 if ((float)hash_table->amount / hash_table->capacity >
243 RSV_HASH_TABLE_LOAD_FACTOR) {
244 rsv_hash_table_resize(hash_table, hash_table->capacity * 2);
245 }
246
247 index = hash_table->custom_hash_func(key, hash_table->key_size) %
248 hash_table->capacity;
249 entry = hash_table->data[index];
250
251 while (entry) {
252 if (hash_table->custom_compare_func(entry->key, key,
253 hash_table->key_size)) {
254 memcpy(entry->value, value, hash_table->value_size);
255 return;
256 }
257 entry = entry->next;
258 }
259
260 new_entry = (rsv_hash_table_entry_t*)malloc(sizeof(rsv_hash_table_entry_t));
261 new_entry->key = malloc(hash_table->key_size);
262 new_entry->value = malloc(hash_table->value_size);
263 memcpy(new_entry->key, key, hash_table->key_size);
264 memcpy(new_entry->value, value, hash_table->value_size);
265 new_entry->next = hash_table->data[index];
266 hash_table->data[index] = new_entry;
267 hash_table->amount++;
268}

276static inline void rsv_hash_table_pop(rsv_hash_table_t* hash_table,
277 const void* key) {
278 unsigned int index = hash_table->custom_hash_func(key, hash_table->key_size) %
279 hash_table->capacity;
280 rsv_hash_table_entry_t* entry = hash_table->data[index];
281 rsv_hash_table_entry_t* prev = NULL;
282
283 while (entry) {
284 if (hash_table->custom_compare_func(entry->key, key,
285 hash_table->key_size)) {
286 if (prev) {
287 prev->next = entry->next;
288 } else {
289 hash_table->data[index] = entry->next;
290 }
291
292 free(entry->key);
293 free(entry->value);
294 free(entry);
295 hash_table->amount--;
296
297 return;
298 }
299
300 prev = entry;
301 entry = entry->next;
302 }
303}
